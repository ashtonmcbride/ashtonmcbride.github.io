Description

This project was the final project to the Intro to C++ Programming course that I completed in the Spring of 2024. It is a multi-semester project inspired by a NSF/NSA grant funded research study conducted in partnership with Texas A&M University. For this project, students were asked to create a C++ program designed entirely by themselves.

Scenario:
I am the CEO of a new and upcoming company. I will be performing a live demo in front of a panel of investors with lots of money. Based on how well I can pitch my new program, I will receive up to 100 thousand dollars to fund my project. Even if I do not receive the entire 100 thousand dollars, the investors are willing to give partial funding for successfully accomplishing major milestones (ie. providing design documents, a spellchecked slide show, and a sales pitch video).

//  ITSE-1307-001 Final Programming Project
//  Author: Ashton McBride
//  Date: 5/11/2024
//  DroneSphere - main.cpp

#include <iostream>
#include <string>
#include <iomanip>
using namespace std;

// Drone Class
class Drone {
private:
    string name;
    string task;
public:

    // Constructor
    Drone(const string& n) : name(n), task("") {
        cout << name << " Connected!" << endl;
        cout << name << " Powering Up..." << endl;
    }

    // Getters
    string getName() {
        return name;
    }

    string getTask() {
        return task;
    }

    // Setters
    void setName(const string& n) {
        name = n;
    }

    void setTask(const string& t) {
        task = t;
    }

    // Function: Displays Task
    void performTask() const {
        cout << task << endl;
    }

};

// Struct to hold a Command
struct CommandAction {
    string command;
    string action;
};

// Function Prototypes
void processDroneCommand(Drone& drone, const string& command);



// Main Function
int main() {

    // Declare Variables
    string userInput = "";

    // Display Message
    cout << setfill('-') << setw(60) << "-" << endl;
    cout << "Welcome to Dronesphere, the future of drone technology!" << endl;
    cout << "This is the virtual command program used to control your drone." << endl << endl;
    cout << "Type 'Exit' to shut down the Drone." << endl;
    cout << setw(60) << "-" << endl;

    // Create a Drone Object
    Drone myDrone("Drone");
	
    // Main Loop
    while (userInput != "Exit")
    {
        // Get User Input
        cout << "\nWhat task would you like the drone to perform?" << endl;
        getline(cin, userInput);

        if (userInput == "Exit")
        {
            // Exit the Program
            cout << "Shutting Down..." << endl;
            break;
        }
        else
        {
            // Process User Command
            processDroneCommand(myDrone, userInput);
        }

    }

    return 0;
}



// Function #1: Processes User Commands for a Drone
void processDroneCommand(Drone& drone, const string& command)
{
    // Map of All Possible Commands
    CommandAction droneActions[] = {
         {"Clean up", "Performing environmental cleanup..."},
         {"Surveillance", "Conducting surveillance..."},
         {"Send delivery", "Delivering package..."},
         {"Pick up delivery", "Picking up package..."},
         {"Take pictures", "Taking pictures..."},
         {"Inspect infrastructure", "Inspecting infrastructure..."},
         {"Plant crops", "Plant crops..."},
    };

    // Check if Command Exists By Looping Through List
    bool commandFound = false;

    for (const auto& action : droneActions)
    {
        if (action.command == command)
        {
            drone.setTask(action.action);
            drone.performTask();
            commandFound = true;
            break;
        }
    }
    if (!commandFound)
    {
        cout << "Command Not Recognized!" << endl;
    }
}
