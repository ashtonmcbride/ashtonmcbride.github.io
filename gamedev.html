// Function #2: Displays a complete list of tasks that the user created
void DisplayTasks(vector<Task>& taskList, vector<Category>& categoryList)
{
    cout << "\n>> TASK LIST <<" << endl;
    cout << "---------------------------------------------------------" << endl;
    cout << "Here is a list of all of your existing tasks:" << endl << endl;

    for (int i = 0; i < taskList.size(); i++)
    {
        cout << taskList[i].GetTaskId() << ". " << taskList[i].GetTaskName() << " | Due: " << taskList[i].GetTaskDueDate()
             << " | Reminder(s): " << taskList[i].GetTaskReminder() << " | Priority: " << taskList[i].GetTaskPriority()
             << " | Category: " << ((taskList[i].GetCategoryId() == 0) ? "None" : categoryList[taskList[i].GetCategoryId() - 1].GetCategoryName())
             << " | Status: " << taskList[i].GetTaskStatus()
             << endl;
    }

    cout << "---------------------------------------------------------" << endl;
}


// Function #3: Displays a complete list of categories that the user created
void DisplayCategories(vector<Category>& categoryList)
{
    cout << "\n>> CATEGORY LIST <<" << endl;
    cout << "---------------------------------------------------------" << endl;
    cout << "Here is a list of all of the existing categories:" << endl << endl;

    for (Category category : categoryList)
    {
        cout << category.GetCategoryId() << ". " << category.GetCategoryName()
             << " | Tasks: " << category.GetCategoryTaskAmount() << endl;
    }

    cout << "---------------------------------------------------------" << endl;
}


// Function #4: Reassigns task IDs and puts them in correct order after a task is deleted
void ReassignTaskIDs(vector<Task>& taskList)
{
    for (int i = 0; i < taskList.size(); i++)
    {
        taskList[i].SetTaskId(i + 1);
    }
}


// Function #5: Reassigns category IDs and puts them in correct order + updates task category IDs after a category is deleted
void ReassignCategoryIDs(vector<Category>& categoryList, vector<Task>& taskList)
{
    for (int i = 0; i < taskList.size(); i++)
    {
        if (taskList[i].GetCategoryId() != 0)
        {
            bool categoryFound = false;

            for (int j = 0; j < categoryList.size(); j++)
            {
                if (taskList[i].GetCategoryId() == categoryList[j].GetCategoryId())
                {
                    taskList[i].SetCategoryId(j + 1);
                    categoryFound = true;
                    break;
                }
            }

            if (categoryFound == false)
            {
                taskList[i].SetCategoryId(0);
            }
        }
    }

    for (int i = 0; i < categoryList.size(); i++)
    {
        categoryList[i].SetCategoryId(i + 1);
    }
}


// Function #6: Loads saved tasks from a text document and adds them to the taskList vector
void LoadTasks(vector<Task>& taskList, int& taskCount)
{
    ifstream taskSaves("Tasks.txt");
	
    if (taskSaves.is_open())
    {
        taskList.clear();
        string line;

        while (getline(taskSaves, line))
        {
            stringstream s(line);
            string taskId, name, dueDate, reminder, priorityLevel, categoryId, status;

            if (line.empty())
            {
                cout << "Empty line!" << endl;
            }
            else
            {
                getline(s, taskId, '|');
                getline(s, name, '|');
                getline(s, dueDate, '|');
                getline(s, reminder, '|');
                getline(s, priorityLevel, '|');
                getline(s, categoryId, '|');
                getline(s, status, '|');
				
                Task loadedTask;
				
                loadedTask.SetTaskId(stoi(taskId));
                loadedTask.SetTaskName(name);
                loadedTask.SetTaskDueDate(dueDate);
                loadedTask.SetTaskReminder(reminder);
                loadedTask.SetTaskPriority(priorityLevel);
                loadedTask.SetCategoryId(stoi(categoryId));
                loadedTask.SetTaskStatus(status);
				
                taskList.push_back(loadedTask);
                taskCount = taskCount + 1;
            }
        }

        // cout << "Successfully loaded all tasks!" << endl;
        taskSaves.close();
    }
    else
    {
        cerr << "Error: Failed to open Tasks file." << endl;
        return;
    }
}

// Function #8: Saves all created tasks to a text document
void SaveTasks(vector<Task>& taskList)
{
    ofstream taskSaves("Tasks.txt");
	
    if (taskSaves.is_open())
    {
        for (int i = 0; i < taskList.size(); i++)
        {
            taskSaves << taskList[i].GetTaskId() << "|"
                      << taskList[i].GetTaskName() << "|"
                      << taskList[i].GetTaskDueDate() << "|"
                      << taskList[i].GetTaskReminder() << "|"
                      << taskList[i].GetTaskPriority() << "|"
                      << taskList[i].GetCategoryId() << "|"
                      << taskList[i].GetTaskStatus() << endl;
        }

        cout << "Successfully saved all tasks!" << endl;
        taskSaves.close();
    }
    else
    {
        cerr << "Error: Failed to open Tasks file." << endl;
        return;
    }
}

}
